Adding properties

Dot notation

                    
                        const person = {
                            name: 'John',
                            age: 24
                        };

                        person.lastName = 'Hero';

Adding properties

Bracket notation

                    
                        const person = {
                            name: 'John',
                            age: 24
                        };

                        person['lastName'] = 'Hero';

                        // Useful when you have the key in variable
                        const key = 'sex';
                        person[key] = 'male';

Deleting properties

Dot notation

                    
                        const person = {
                            name: 'John',
                            age: 24
                        };

                        delete person.name;

                        console.log(person); // {age: 24}

Deleting properties

Bracket notation

                    
                        const person = {
                            name: 'John',
                            age: 24
                        };

                        delete person['age'];

                        console.log(person); // {name: 'John'}

Accessing Properties

Dot notation

                    
                        const john = {
                            name: 'John',
                            lastName: 'Hero',
                            age: 24
                        };

                        console.log(john.age); // 24
                        console.log(john.lastName); // Hero
                        console.log(john.middleName); // undefined

Accessing Properties

Bracket notation

                    
                        const john = {
                            name: 'John',
                            lastName: 'Hero',
                            age: 24
                        };

                        console.log(john['age']); // 24
                        console.log(john['lastName']); // Hero
                        console.log(john['middleName']; // undefined

Accessing Properties

Destructuring

                    
                        const john = {
                            name: 'John',
                            age: 24
                        };

                        const {name, age} = john;

                        console.log(name); // John
                        console.log(age); // 24

                        const {name: foo, age: bar} = john;

                        console.log(foo); // John
                        console.log(bar); // 24

Accessing Properties

Destructuring - Default Values

                    
                        const john = {
                            name: 'John',
                            age: 24
                        };

                        const {name, lastName = 'Hero'} = john;

                        console.log(name); // John
                        console.log(lastName); // Hero

                        const {name: foo, lastName: bar = 'Hero'} = john;

                        console.log(foo); // John
                        console.log(bar); // Hero

Accessing Properties

Destructuring - Computed Keys

                    
                        const john = {
                            name: 'John',
                            age: 24
                        };

                        const key = 'name';

                        const {[key]: foo} = john;

                        console.log(foo); // John
